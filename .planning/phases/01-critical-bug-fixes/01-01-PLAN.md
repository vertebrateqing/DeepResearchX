```python
def _merge_retrieval_results(all_results: list[list[dict[str, Any]]]) -> list[dict[str, Any]]:
    seen: dict[str, dict[str, Any]] = {}
    for results in all_results:
        for doc in results:
            doc_id = doc.get("id", "")
            score = doc.get("score", 0)
            if doc_id in seen:
                if score < seen[doc_id]["score"]:
                    seen[doc_id] = doc
            else:
                seen[doc_id] = doc
    merged = sorted(seen.values(), key=lambda d: d.get("score", float("inf")))
    return merged
```

    ```python
                [{"id": "doc1", "score": 0.9, "content": "a"}, {"id": "doc2", "score": 0.5, "content": "b"}],

            assert doc1["score"] == 0.9  # higher similarity wins
            # Should be sorted descending by score
            scores = [d["score"] for d in merged]
            assert scores == sorted(scores, reverse=True)

        def test_merge_empty(self):

            assert _merge_retrieval_results([]) == []

            assert _merge_retrieval_results(results) == []
    ```
